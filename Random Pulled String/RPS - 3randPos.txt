def setup():
    size(400, 400)
    colorMode(RGB)
    noStroke()
    background(0,0,0)
    strokeWeight(10)
i = 0
def draw():
    global i
    if i < width:
        i = i + 1
        r = random(255)
        x = random(0,width)
        stroke(r,r,r,100)
        line(i,0,x,height)
    if mousePressed:
        i = random(0,width)
