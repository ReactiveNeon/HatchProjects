def setup():
    size(400, 400)
    colorMode(RGB)
    noStroke()
    background(0,0,0)
    strokeWeight(10)
i = 0
a = 1
def draw():
   global i, a

   if i == 400:
       a = -1
   elif i == 0:
       a = 1   
   i = i + a
   r = random(255)
   x = random(0,width)
   stroke(r,r,r,100)
   line(i,0,x,height)
