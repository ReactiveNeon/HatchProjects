def setup():
    size(600, 600)
    colorMode(RGB)
i = 0
def mouseClicked():
    fill(random(0,255),random(0,255),random(0,255))
def draw():
    global i 
    mouseDist = dist(300,300,mouseX,mouseY)
    resetMatrix()
    translate(300,300)
    rotate(i*20)
    ellipse(0,mouseDist,mouseDist/1.25,mouseDist/1.25)
    i = i+1
