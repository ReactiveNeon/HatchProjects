def setup():
    size(600, 600)
    colorMode(RGB)
    noStroke()
i = 0
def mouseClicked():
    fill(random(0,255),random(0,255),random(0,255))
def draw():
    global i 
    mouseDist = dist(300,300,mouseX,mouseY)
    resetMatrix()
    translate(300,300)
    rotate(i*20)
    fill(mouseX/2,mouseY/2,(mouseX+mouseY)/4)
    rect(0,mouseDist,mouseDist/1.25,mouseDist/1.25)
    i = i+1
